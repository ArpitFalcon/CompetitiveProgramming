struct Trie{
    bool end;
    Trie *children[10];
};

Trie* root;

Trie* createNode(){
    Trie* node = new Trie();
    for(int i=0;i<10;i++){
        node -> children[i] = NULL;
    }
    return node;
}

bool search(string s){
    if(root==NULL){
        return false;
    }
    Trie* curr = root;
    for(int i=0;i<s.size();i++){
        char c = s[i];
        int index = c - '0';

        if(curr-> children[index] == NULL){
            return false;
        }
        curr = curr -> children[index];
    }
    return (curr != NULL && curr -> end == true);
}

void insert(string s){
    if(root == NULL){
        root = createNode();
    }
    Trie* curr = root;
    for(int i=0;i<s.size();i++){
        char c = s[i];
        int index = c - '0';
        if(curr-> children[index] == NULL){
            curr -> children[index] = createNode();
        }
        curr = curr -> children[index];
    }
    curr -> end = true;
}
