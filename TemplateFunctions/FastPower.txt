long power(int a, int N){
    if(N==0)
        return 1;
    else if(N==1)
        return a;
    else{
        int R = power(a ,N/2);
        if(N%2==0)
            return R*R;
        else
            return R*a*R;
    }
}

long power(int a, int N, int M) {
    if (N == 0) {
        return 1;
    } else {
        long R = power(a, N / 2, M);
        if (N % 2 == 0) {
            return (R * R) % M;
        } else {
            return (R * R * a) % M;
        }
    }
}
